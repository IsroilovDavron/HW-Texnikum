#'Power rangers'

print('Третий символ строки:', a[2])
print('Предпоследний символ строки:', a[-2])
print('Первые пять символов строки:', a[0:5])
print('Все символы кроме последних двух:', a[0:-2])
print('Символы по четным индексам:', a[::2]) 
print('Символы по нечетным индексам:', a[1::2]) 
print('Символы в обратном порядке', a[::-1]) 
print('Символы в обратном порядке чередами:', a[::-2]) 
print('Длина строки:', len(a))
