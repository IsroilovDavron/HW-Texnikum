#'Матемагия'
print('Итак, нам надо докозать что равенство\n"1+2+...+n=n(n+1)/2" выполняется')
n=int(input('Для этого введите любое натуральное число: '))
i=0
s1=0
while i<n:
  i=i+1
  s1=s1+i
s2=int((n/2)*(n+1))
print('Cумма числе от 1 до',n,'будет равна:', s1)
print('Согласно формуле n(n+1)/2, результат будет равен:', s2)
if s1==s2:
  print('\nВот и результат',s1,'=',s2 )
